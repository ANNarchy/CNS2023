Maith, O., Escudero, F. V., Dinkelbach, H. Ü., Baladron, J., Horn, A., Irmen, F., et al. 
A computational model-based analysis of basal ganglia pathway changes in Parkinson’s disease inferred from resting-state fMRI.
European Journal of Neuroscience. doi:10.1111/ejn.14868

BOLD model definition

balloon_maith2021 = BoldModel(
    parameters="""
            second    = 1000.0
            phi   = 1.0
            kappa = 0.665
            gamma = 0.412
            E_0       = 0.3424
            tau     = 1.0368
            alpha     = 0.3215
            V_0       = 0.02
            noise = 0
        """,
    equations="""
            I_CBF          = sum(I_CBF) + noise                                         : init=0
            ds/dt          = (phi * I_CBF - kappa * s - gamma * (f_in - 1))/second      : init=0
            df_in/dt       = s / second                                                 : init=1, min=0.01

            E              = 1 - (1 - E_0)**(1 / f_in)                                  : init=0.3424
            dq/dt          = (f_in * E / E_0 - (q / v) * f_out)/(tau*second)            : init=1, min=0.01
            dv/dt          = (f_in - f_out)/(tau*second)                                : init=1, min=0.01
            f_out          = v**(1 / alpha)                                             : init=1, min=0.01

            k_1            = 7 * E_0
            k_2            = 2
            k_3            = 2 * E_0 - 0.2

            BOLD           = V_0 * (k_1 * (1 - q) + k_2 * (1 - (q / v)) + k_3 * (1 - v)) : init=0
        """,
    inputs="I_CBF",
    output="BOLD"
)

Neuron model definitions

Hybrid_neuron = Neuron(
    parameters="""
    a = 0.0   
    b = 0.0   
    c = 0.0   
    d = 0.0   
    n0 = 140. 
    n1 = 5.0  
    n2 = 0.04 
    I = 0.0   
    tau_ampa = 10 
    tau_gaba = 10 
    E_ampa = 0.0 
    E_gaba = -90.0 
    tau_syn = 1.0
    SteadyStateNoise = 0.0
    baseline = 0.0
    tau_noise=200.
""",
    equations="""
    dg_ampa/dt = -g_ampa/tau_ampa : init = 0
    dg_gaba/dt = -g_gaba/tau_gaba : init = 0
    dslowNoise/dt = (-slowNoise+SteadyStateNoise)/tau_noise : init = 0
    dv/dt = n2*v*v+n1*v+n0 - u  + I - g_ampa*(v-E_ampa) - g_gaba*(v-E_gaba) + slowNoise + baseline : init = -70.
    du/dt = a*(b*(v)-u) : init = -18.55
    tau_syn*dsyn/dt = -syn 
""",
    spike="""
    v>=30
""",
    reset="""
    v = c
    u = u+d
""",
    refractory=10.0,
)

def simulateNoisePeriods(net, duration, rng):
    try:
        if duration % noiseFrequency != 0:
            raise ValueError(
                "duration ("
                + str(duration)
                + ") has to be a multiple of noiseFrequency"
            )
    except ValueError as ve:
        print(ve)
        exit(1)
    for NoisePeriod in range(int(duration / noiseFrequency)):
        net.get_population("V1").SteadyStateNoise = rng.normal(
            Cortex_noise_mean + rng.uniform(-Cortex_noise_delta, Cortex_noise_delta),
            Cortex_noise_sd,
            net.get_population("V1").geometry,
        )
        net.get_population("SD1").SteadyStateNoise = rng.normal(
            SD1_noise_mean + rng.uniform(-SD1_noise_delta, SD1_noise_delta),
            SD1_noise_sd,
            net.get_population("SD1").geometry,
        )
        net.get_population("SD2").SteadyStateNoise = rng.normal(
            SD2_noise_mean + rng.uniform(-SD2_noise_delta, SD2_noise_delta),
            SD2_noise_sd,
            net.get_population("SD2").geometry,
        )
        net.get_population("GPi").SteadyStateNoise = rng.normal(
            GPi_noise_mean + rng.uniform(-GPi_noise_delta, GPi_noise_delta),
            GPi_noise_sd,
            net.get_population("GPi").geometry,
        )
        net.get_population("GPe").SteadyStateNoise = rng.normal(
            GPe_noise_mean + rng.uniform(-GPe_noise_delta, GPe_noise_delta),
            GPe_noise_sd,
            net.get_population("GPe").geometry,
        )
        net.get_population("STN").SteadyStateNoise = rng.normal(
            STN_noise_mean + rng.uniform(-STN_noise_delta, STN_noise_delta),
            STN_noise_sd,
            net.get_population("STN").geometry,
        )
        net.get_population("Thalamus").SteadyStateNoise = rng.normal(
            Thalamus_noise_mean
            + rng.uniform(-Thalamus_noise_delta, Thalamus_noise_delta),
            Thalamus_noise_sd,
            net.get_population("Thalamus").geometry,
        )

        net.get(m_bold_GPi).noise = rng.uniform(0, 1, 1) * noiseSYAC
        net.get(m_bold_GPe).noise = rng.uniform(0, 1, 1) * noiseSYAC
        net.get(m_bold_STN).noise = rng.uniform(0, 1, 1) * noiseSYAC
        net.get(m_bold_Strboth).noise = rng.uniform(0, 1, 1) * noiseSYAC
        net.get(m_bold_Thal).noise = rng.uniform(0, 1, 1) * noiseSYAC
        net.get(m_bold_Cortex).noise = rng.uniform(0, 1, 1) * noiseSYAC
        net.simulate(noiseFrequency)

Analysis: the loss (difference between simulated and recorded BOLD response) depends on the simulation length

bold_arr_len = len(bold_gpe_arr)
time_step_list = list(
    range(int(bold_arr_len / 1000), bold_arr_len, int(bold_arr_len / 1000))
)
cor_list = []
for idx in time_step_list:
    cor_list.append(
        np.corrcoef(
            [
                bold_gpe_arr[:idx],
                bold_gpi_arr[:idx],
                bold_stn_arr[:idx],
                bold_cor_arr[:idx],
                bold_str_arr[:idx],
                bold_tha_arr[:idx],

        )
    )
